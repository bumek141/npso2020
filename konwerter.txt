    }

}

int liczenie (int x, int y){

    int a[256],j,i=0;

    while (x>0){
        y=(x%2);
        a[i++]=y;
        x=x/2;
    }
    for(j=i-1;j>=0; j--){
        printf("%i",a[j]);
    }

    return y;
}
